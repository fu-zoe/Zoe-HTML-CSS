Chapter 2 <head>

The <head> Element

Metadata, styles, icons, performance hints, alternates, and social preview. Mostly code — copy, paste, and tweak.

Minimal document (with <head>)

<!doctype html>
<html lang="en">
  <head>
    <!-- Character encoding -->
    <meta charset="UTF-8">

    <!-- Responsive viewport -->
    <meta name="viewport" content="width=device-width, initial-scale=1">

    <!-- Author & short description (used by search/snippets) -->
    <meta name="author" content="Zoe Fu">
    <meta name="description" content="This page contains all the things I am learning as I learn HTML">

    <!-- Title (browser tab) -->
    <title>My First Web Page</title>

    <!-- Favicon -->
    <link rel="icon" href="html-5.png" type="image/x-icon">

    <!-- Stylesheet (type is optional in HTML5) -->
    <link rel="stylesheet" href="main.css">
  </head>
  <body>
  </body>
</html>

Common <link> rel reference

<!-- CSS stylesheet -->
<link rel="stylesheet" href="main.css">
<!-- Optional: media-specific styles -->
<link rel="stylesheet" href="print.css" media="print">

<!-- Favicon / app icons -->
<link rel="icon" href="favicon.ico" type="image/x-icon">
<link rel="icon" href="favicon.png" type="image/png">
<link rel="apple-touch-icon" href="apple-touch-icon.png" sizes="180x180">
<link rel="manifest" href="/site.webmanifest">

<!-- Preload critical resources -->
<link rel="preload" href="main.css" as="style">
<link rel="preload" href="app.js" as="script">

<!-- Preconnect / DNS-prefetch (network hints) -->
<link rel="preconnect" href="https://fonts.gstatic.com" crossorigin>
<link rel="preconnect" href="https://fonts.googleapis.com">
<link rel="dns-prefetch" href="//example-cdn.com">

<!-- Alternate versions (feeds, translations) -->
<link rel="alternate" href="feed.xml" type="application/rss+xml" title="RSS">
<link rel="alternate" href="page-fr.html" hreflang="fr">

<!-- Canonical URL (avoid duplicate content) -->
<link rel="canonical" href="https://example.com/page">

<!-- Module preload (for ESM) -->
<link rel="modulepreload" href="main.mjs">

Meta tags — quick cookbook

<!-- Basics -->
<meta charset="UTF-8">
<meta name="viewport" content="width=device-width, initial-scale=1">
<meta name="author" content="Zoe Fu">
<meta name="description" content="Short description for search/snippets.">

<!-- Robots (indexing control) -->
<meta name="robots" content="index,follow">

<!-- Theme color (browser UI tint on mobile) -->
<meta name="theme-color" content="#6fb0d8">

<!-- Open Graph (social preview) -->
<meta property="og:title" content="My First Web Page">
<meta property="og:description" content="A simple demo of HTML head tags.">
<meta property="og:type" content="website">
<meta property="og:url" content="https://example.com">
<meta property="og:image" content="https://example.com/preview.png">

<!-- Twitter Card (summary preview) -->
<meta name="twitter:card" content="summary_large_image">
<meta name="twitter:title" content="My First Web Page">
<meta name="twitter:description" content="A simple demo of HTML head tags.">
<meta name="twitter:image" content="https://example.com/preview.png">

Stylesheets — loading patterns

<!-- Regular blocking stylesheet (simple & reliable) -->
<link rel="stylesheet" href="main.css">

<!-- Print-only CSS -->
<link rel="stylesheet" href="print.css" media="print">

<!-- Optional: preload + swap (advanced; test before using) -->
<link rel="preload" href="critical.css" as="style" onload="this.onload=null;this.rel='stylesheet'">
<noscript><link rel="stylesheet" href="critical.css"></noscript>

Scripts in <head> (only if needed)

<!-- Defer classic scripts so they don't block HTML parsing -->
<script src="app.js" defer></script>

<!-- ES modules are deferred by default -->
<script type="module" src="main.mjs"></script>

<!-- Inline critical config (keep small) -->
<script>window.__CONFIG__ = { lang: 'en' };</script>
